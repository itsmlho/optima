# ================================================================
# SSH COMMANDS - Copy Paste Step by Step
# ================================================================

# STEP 1: Find optima application
ls -la domains/
# Cari folder domain yang berisi aplikasi optima

# STEP 2: Navigate ke domain folder (ganti [DOMAIN] dengan hasil step 1)
cd domains/[DOMAIN]/public_html
pwd
ls -la

# STEP 3: Check if this is optima application
ls -la app/Controllers/Marketing.php
ls -la spark
# Jika file-file ini ada, berarti ini aplikasi optima

# STEP 4: Check database credentials
cat .env | grep -E "database.default.database|database.default.username|database.default.password|database.default.hostname"

# STEP 5: Test database connection (gunakan credentials dari step 4)
mysql -u [DB_USER] -p -h [DB_HOST]
# Enter password saat diminta
# Setelah masuk MySQL prompt:

SHOW DATABASES;
USE [DATABASE_NAME];
SHOW TABLES LIKE '%delivery%';
SHOW TABLES LIKE '%instruksi%';

# STEP 6: Check actual table structure
DESCRIBE delivery_instructions;
# Atau jika error, coba:
SHOW TABLES;

# STEP 7: Exit MySQL
exit;
